<script>
  const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbxBSKtA-RW4vy7HPqFairWgDHyG-1s2PkjpKWCtR_yc5anTt2AxaOz_hIaA96U27c0U/exec';
  // ─────────────────────────────────────────────
  // DYNAMIC FIELDS
  // ─────────────────────────────────────────────
  function updateRemoveButtons() {
    const skillItems = document.querySelectorAll('#skillContainer .dynamic-field-item');
    skillItems.forEach((item, i) => {
      const btn = item.querySelector('.remove-skill');
      if (btn) btn.style.display = skillItems.length > 1 ? 'inline-block' : 'none';
    });
    const langItems = document.querySelectorAll('#langContainer .dynamic-field-item');
    langItems.forEach((item, i) => {
      const btn = item.querySelector('.remove-lang');
      if (btn) btn.style.display = langItems.length > 1 ? 'inline-block' : 'none';
    });
  }

  document.getElementById('addSkillBtn')?.addEventListener('click', () => {
    const container = document.getElementById('skillContainer');
    const div = document.createElement('div');
    div.className = 'dynamic-field-item';
    div.innerHTML = `<input type="text" placeholder="เช่น JavaScript" class="skill-input"><button type="button" class="remove-skill">✕</button>`;
    container.appendChild(div);
    div.querySelector('.remove-skill').addEventListener('click', () => {
      div.remove();
      updateRemoveButtons();
    });
    updateRemoveButtons();
  });

  document.getElementById('addLangBtn')?.addEventListener('click', () => {
    const container = document.getElementById('langContainer');
    const div = document.createElement('div');
    div.className = 'dynamic-field-item';
    div.innerHTML = `<input type="text" placeholder="เช่น อังกฤษ" class="lang-input"><button type="button" class="remove-lang">✕</button>`;
    container.appendChild(div);
    div.querySelector('.remove-lang').addEventListener('click', () => {
      div.remove();
      updateRemoveButtons();
    });
    updateRemoveButtons();
  });

  document.querySelectorAll('.remove-skill').forEach(btn => btn.addEventListener('click', function() {
    this.closest('.dynamic-field-item')?.remove();
    updateRemoveButtons();
  }));
  document.querySelectorAll('.remove-lang').forEach(btn => btn.addEventListener('click', function() {
    this.closest('.dynamic-field-item')?.remove();
    updateRemoveButtons();
  }));
  updateRemoveButtons();

  // zone button
  document.getElementById('allZoneBtn')?.addEventListener('click', () => {
    document.getElementById('zone').value = 'กทม. (ทุกเขต)';
  });

  // counters
  const idoField = document.getElementById('ido');
  const descField = document.getElementById('desc');
  if (idoField) idoField.addEventListener('input', () => {
    document.getElementById('idoCounter').textContent = idoField.value.length;
  });
  if (descField) descField.addEventListener('input', () => {
    document.getElementById('descCounter').textContent = descField.value.length;
  });

  // validate URL
  function isValidUrl(url) {
    if (!url) return true;
    return url.startsWith('http://') || url.startsWith('https://');
  }

  function showFieldError(elementId, msg) {
    const err = document.getElementById(elementId);
    if (err) {
      err.textContent = msg;
      err.style.display = 'block';
      setTimeout(() => err.style.display = 'none', 4000);
    }
  }

  // generate edit code
  function generateEditCode() {
    const chars = 'abcdefghjkmnpqrstuvwxyz23456789';
    let code = '';
    for (let i = 0; i < 8; i++) {
      code += chars[Math.floor(Math.random() * chars.length)];
    }
    return code;
  }

  // submit form
  document.getElementById('createForm')?.addEventListener('submit', async (e) => {
    e.preventDefault();

    const firstname = document.getElementById('firstname').value.trim();
    const lastname = document.getElementById('lastname').value.trim();
    const photo = document.getElementById('photo').value.trim();
    const ido = document.getElementById('ido').value.trim();
    const desc = document.getElementById('desc').value.trim();
    const exp = document.getElementById('exp').value.trim();
    const salary = document.getElementById('salary').value;
    const available = document.getElementById('available').value.trim();
    const zone = document.getElementById('zone').value.trim();
    const line = document.getElementById('line').value.trim();
    const phone = document.getElementById('phone').value.trim();
    const email = document.getElementById('email').value.trim();
    const ig = document.getElementById('ig').value.trim();
    const tt = document.getElementById('tt').value.trim();
    const resume = document.getElementById('resume').value.trim();

    const selectedCaps = Array.from(document.querySelectorAll('#capabilityGroup input:checked')).map(cb => cb.value);
    const selectedWorkTypes = Array.from(document.querySelectorAll('#workTypeGroup input:checked')).map(cb => cb.value);

    const skillTags = Array.from(document.querySelectorAll('#skillContainer .skill-input')).map(inp => inp.value.trim()).filter(v => v);
    const languages = Array.from(document.querySelectorAll('#langContainer .lang-input')).map(inp => inp.value.trim()).filter(v => v);

    if (!firstname || !lastname || !photo || !ido || !desc || !exp || !salary || !available || !zone || !resume) {
      alert('กรุณากรอกข้อมูลให้ครบทุกช่องที่มี *');
      return;
    }
    if (selectedCaps.length === 0) {
      alert('กรุณาเลือก Capability อย่างน้อย 1 หมวด');
      return;
    }
    if (selectedWorkTypes.length === 0) {
      alert('กรุณาเลือกรูปแบบงานอย่างน้อย 1 แบบ');
      return;
    }
    if (skillTags.length === 0) {
      alert('กรุณากรอกทักษะอย่างน้อย 1 อย่าง');
      return;
    }
    if (languages.length === 0) {
      alert('กรุณากรอกภาษาอย่างน้อย 1 ภาษา');
      return;
    }
    if (!isValidUrl(photo)) {
      showFieldError('photoError', 'URL รูปภาพไม่ถูกต้อง');
      return;
    }
    if (!isValidUrl(resume)) {
      showFieldError('resumeError', 'URL Resume ไม่ถูกต้อง');
      return;
    }

    const instagramUrl = ig ? `https://instagram.com/${ig.replace('@', '')}` : '';
    const tiktokUrl = tt ? `https://tiktok.com/@${tt.replace('@', '')}` : '';

    const editCode = generateEditCode();

    const payload = {
      action: 'create',
      firstname, lastname,
      photo_id: photo,
      i_do_good: ido,
      description: desc,
      purpose_category: selectedCaps.join(', '),
      experience: exp,
      skill_tags: skillTags.join(', '),
      language: languages.join(', '),
      work_type: selectedWorkTypes.join(', '),
      expected_salary: salary,
      available_date: available,
      work_zone: zone,
      contact_line: line,
      contact_phone: phone,
      contact_email: email,
      instagram: instagramUrl,
      tiktok: tiktokUrl,
      resume: resume,
      edit_code: editCode,
      status: 'TRUE'
    };

    console.log('Sending payload:', payload);

    try {
      const response = await fetch(SCRIPT_URL, {
        method: 'POST',
        mode: 'no-cors',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });
  
      // เนื่องจาก mode: 'no-cors' จะอ่าน response ไม่ได้
      // ให้แสดง Popup ทันที (สมมติว่าสำเร็จ)
      document.getElementById('editCodeDisplay').textContent = editCode;
      document.getElementById('popupOverlay').classList.add('open');
  
      document.getElementById('copyCodeBtn').onclick = () => {
        navigator.clipboard.writeText(editCode);
        alert('คัดลอกรหัสแล้ว');
      };
      document.getElementById('backHomeBtn').onclick = () => {
        window.location.href = 'index.html';
      };
  
    } catch (err) {
      alert('เกิดข้อผิดพลาด: ' + err.message);
    }

    // TODO: ส่งไป Google Apps Script (จะแยกส่งทีหลัง)
    // แสดง Popup
    document.getElementById('editCodeDisplay').textContent = editCode;
    document.getElementById('popupOverlay').classList.add('open');

    // ปุ่ม copy
    document.getElementById('copyCodeBtn').onclick = () => {
      navigator.clipboard.writeText(editCode);
      alert('คัดลอกรหัสแล้ว');
    };
    document.getElementById('backHomeBtn').onclick = () => {
      window.location.href = 'index.html';
    };
  });
</script>

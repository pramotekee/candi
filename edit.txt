  function renderEditSkillTags(tagsArray) {
    const container = document.getElementById('editSkillContainer');
    if (!container) return;
    container.innerHTML = '';
    tagsArray.forEach((tag, idx) => {
      const div = document.createElement('div');
      div.className = 'dynamic-field-item';
      div.innerHTML = `
        <input type="text" class="edit-skill-input" value="${escapeHtml(tag)}" placeholder="เช่น Python">
        <button type="button" class="remove-edit-skill">✕</button>
      `;
      container.appendChild(div);
      div.querySelector('.remove-edit-skill').addEventListener('click', () => {
        div.remove();
      });
    });
    if (tagsArray.length === 0) addEditSkillField();
  }

  function escapeHtml(str) {
    if (!str) return '';
    return str.replace(/[&<>]/g, function(m) {
      if (m === '&') return '&amp;';
      if (m === '<') return '&lt;';
      if (m === '>') return '&gt;';
      return m;
    });
  }

  // ─────────────────────────────────────────────
  // SEARCH PROFILE (จะเชื่อม Google Sheets)
  // ─────────────────────────────────────────────
  async function searchProfile() {
    const editCode = document.getElementById('editCode').value.trim();
    const firstname = document.getElementById('searchFirstname').value.trim();
    const lastname = document.getElementById('searchLastname').value.trim();

    if (!editCode || !firstname || !lastname) {
      showAlert('กรุณากรอกข้อมูลให้ครบทั้ง 3 ช่อง', 'error');
      return;
    }

    showAlert('กำลังค้นหา...', 'info');

    try {
      const response = await fetch(SCRIPT_URL, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          action: 'fetch',
          edit_code: editCode,
          firstname: firstname,
          lastname: lastname
        })
      });
  
      const result = await response.json();
  
      if (result.success && result.data) {
        fillEditForm(result.data);
        currentEditCode = editCode;
        currentRowId = result.data.id;
        document.getElementById('editSection').classList.add('show');
        showAlert('พบโปรไฟล์แล้ว กรอกข้อมูลที่ต้องการแก้ไข', 'success');
      } else {
        showAlert(result.error || 'ไม่พบโปรไฟล์ ตรวจสอบรหัส ชื่อ และนามสกุลอีกครั้ง', 'error');
      }
    } catch (err) {
      showAlert('เกิดข้อผิดพลาด: ' + err.message, 'error');
    }
  }

  function fillEditForm(data) {
    document.getElementById('editFirstname').value = data.firstname || '';
    document.getElementById('editLastname').value = data.lastname || '';
    document.getElementById('editPhoto').value = data.photo_id || '';
    document.getElementById('editIdo').value = data.i_do_good || '';
    document.getElementById('editDesc').value = data.description || '';
    document.getElementById('editExp').value = data.experience || '';
    document.getElementById('editSalary').value = data.expected_salary || '';
    document.getElementById('editAvailable').value = data.available_date || '';
    document.getElementById('editZone').value = data.work_zone || '';
    document.getElementById('editLine').value = data.contact_line || '';
    document.getElementById('editPhone').value = data.contact_phone || '';
    document.getElementById('editEmail').value = data.contact_email || '';
    document.getElementById('editResume').value = data.resume || '';

    // Instagram/TikTok ดึง username จาก URL
    const igUrl = data.instagram || '';
    const ttUrl = data.tiktok || '';
    const igUser = igUrl.replace('https://instagram.com/', '').replace('https://www.instagram.com/', '').replace(/\/$/, '');
    const ttUser = ttUrl.replace('https://tiktok.com/@', '').replace('https://www.tiktok.com/@', '').replace(/\/$/, '');
    document.getElementById('editIg').value = igUser !== 'https://instagram.com/' ? igUser : '';
    document.getElementById('editTt').value = ttUser !== 'https://tiktok.com/@' ? ttUser : '';

    // Capability checkboxes
    const caps = (data.purpose_category || '').split(',').map(c => c.trim());
    document.querySelectorAll('#editCapabilityGroup input').forEach(cb => {
      cb.checked = caps.includes(cb.value);
    });

    // Work type checkboxes
    const workTypes = (data.work_type || '').split(',').map(w => w.trim());
    document.querySelectorAll('#editWorkTypeGroup input').forEach(cb => {
      cb.checked = workTypes.includes(cb.value);
    });

    // Skill tags
    const skills = (data.skill_tags || '').split(',').map(s => s.trim()).filter(s => s);
    renderEditSkillTags(skills);

    // Languages
    const langs = (data.language || '').split(',').map(l => l.trim()).filter(l => l);
    renderEditLanguages(langs);

    // counters
    document.getElementById('editIdoCounter').textContent = (data.i_do_good || '').length;
    document.getElementById('editDescCounter').textContent = (data.description || '').length;
  }

  function showAlert(msg, type) {

    setTimeout(() => {
      if (container.firstChild) container.innerHTML = '';
    }, 4000);
  }

    const selectedCaps = Array.from(document.querySelectorAll('#editCapabilityGroup input:checked')).map(cb => cb.value);

    if (!firstname || !lastname || !photo || !ido || !desc || !exp || !salary || !available || !zone || !resume) {
      showAlert('กรุณากรอกข้อมูลให้ครบทุกช่องที่มี *', 'error');
      return;
    }
    if (selectedCaps.length === 0) {
      showAlert('กรุณาเลือก Capability อย่างน้อย 1 หมวด', 'error');
      return;
    }
    if (selectedWorkTypes.length === 0) {
      showAlert('กรุณาเลือกรูปแบบงานอย่างน้อย 1 แบบ', 'error');
      return;
    }
    if (skillTags.length === 0) {
      showAlert('กรุณากรอกทักษะอย่างน้อย 1 อย่าง', 'error');
      return;
    }
    if (languages.length === 0) {
      showAlert('กรุณากรอกภาษาอย่างน้อย 1 ภาษา', 'error');
      return;
    }

    const instagramUrl = ig ? `https://instagram.com/${ig.replace('@', '')}` : '';
    const tiktokUrl = tt ? `https://tiktok.com/@${tt.replace('@', '')}` : '';

    const payload = {
      action: 'update',
      edit_code: currentEditCode,
      firstname, lastname,

  // DELETE PROFILE
  // ─────────────────────────────────────────────
  async function deleteProfile() {
    if (!currentEditCode) {
      showAlert('กรุณาค้นหาโปรไฟล์ก่อน', 'error');
      return;
    }

    const confirmMsg = confirm('⚠️ คุณแน่ใจว่าต้องการลบโปรไฟล์นี้?\nการลบไม่สามารถกู้คืนได้');
    if (!confirmMsg) return;

    const payload = {
      action: 'delete',
      edit_code: currentEditCode,
      firstname: document.getElementById('editFirstname').value.trim(),
      lastname: document.getElementById('editLastname').value.trim()
    };

    console.log('Delete payload:', payload);
    
    try {
      const response = await fetch(SCRIPT_URL, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify(payload)
      });
      
      const result = await response.json();
      
      if (result.success) {
        showAlert('ลบโปรไฟล์สำเร็จ!', 'success');
        setTimeout(() => {
          window.location.href = 'index.html';
        }, 1500);
      } else {
        showAlert(result.error || 'เกิดข้อผิดพลาด', 'error');
      }
    } catch (err) {
      showAlert('เกิดข้อผิดพลาด: ' + err.message, 'error');
    }
    
    // เคลียร์ฟอร์ม
    document.getElementById('editSection').classList.remove('show');
    document.getElementById('editCode').value = '';
    document.getElementById('searchFirstname').value = '';
    document.getElementById('searchLastname').value = '';
    currentEditCode = '';
  }
